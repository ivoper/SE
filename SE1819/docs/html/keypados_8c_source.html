27 void EINT3_IRQHandler(){
28  NVIC_DisableIRQ(EINT3_IRQn);
29  if (!LPC_GPIOINT->IntStatus){ //when initializing there is an interrupt triggered
30  NVIC_EnableIRQ(EINT3_IRQn);
31  return;
32  }
33  LPC_GPIOINT->IO0IntClr |= INPUT_MASK; //clear interrupt
34  BaseType_t TaskWoken = pdFALSE;
35  if(xSemaphoreGiveFromISR(Semaphore, &TaskWoken) == pdPASS){ //resumes keypad task
36  if(TaskWoken == pdTRUE){
37  portYIELD_FROM_ISR(TaskWoken);
38  return;
39  }
40  }
41  xSemaphoreTakeFromISR(Semaphore, &TaskWoken); //if give fails or no thread was waiting, block semaphore again
42  portYIELD_FROM_ISR(TaskWoken);
43  NVIC_EnableIRQ(EINT3_IRQn);
44 }

47 void KEYPADOS_Init(int layout[12], QueueHandle_t queue, SemaphoreHandle_t smphr){
48  Queue = queue;
49  Semaphore = smphr;
50  LPC_PINCON->PINMODE1 &= ~(BITS_TO_PINMODE); //assures input bits are in pull up mode
51  LPC_GPIO0->FIODIR &= ~(INPUT_MASK); //bits for input to 0
52  LPC_GPIO0->FIODIR |= OUTPUT_MASK; //bits for output to 1
53  memcpy(Layout, layout, LAYOUT_SIZE * sizeof(int));
54  LPC_GPIO0->FIOCLR = OUTPUT_MASK; //send all 0s
55  LPC_GPIOINT->IO0IntEnF |= INPUT_MASK; //enable int in input bits
56  NVIC_SetPriority(EINT3_IRQn, 5); //priority to 5, safe to use in freeRTOS interupts
57  LPC_GPIOINT->IO0IntClr |= INPUT_MASK; //just to be sure
58  NVIC_EnableIRQ(EINT3_IRQn);
59 }

70 int KEYPADOS_PoolKeypad(){
71  WAITOS_Ms(30); //for combination of keys
72  int ret = KEY_COMBINATION_UNSUPPORTED_CODE;
73  for(int i=0; i<NUMBER_OF_LINES; ++i){
74  LPC_GPIO0->FIOSETL = OUTPUT_MASK; //output bits to 1
75  LPC_GPIO0->FIOCLRL = 1<<(SHIFT_TO_OUTPUT + i); //line output bit to 0
76  int col = (LPC_GPIO0->FIOPIN & INPUT_MASK)>>SHIFT_TO_INPUT; //input bits(column bits) to col
77  if(col != 7){ //if not all 1s, is the line
78  if ((i == NUMBER_OF_LINES - 1) && (col == 2)){ //combination supported in last line(row) of keypad -> 010
79  ret = COMBINATION_CODE;
80  }
81  if (col == NUMBER_OF_COLUMNS){ //if key is pressed in last column then the col value is correct
82  ret = Layout[col + (i*(NUMBER_OF_LINES -1)) - 1]; //col 3
83  }
84  col ^= VAL_TO_XOR; //if key is pressed in other columns this xor puts correct value in col
85  if (col < NUMBER_OF_COLUMNS){ //can only be col 1 or 2
86  ret = Layout[col + (i*(NUMBER_OF_LINES -1)) - 1];
87  }
88  break; //is the line, but no supported key
89  }
90  }
91  LPC_GPIO0->FIOCLRL = OUTPUT_MASK;
92  return ret;
93 }
