57 int KEYPAD_Read(void){
58  int line = 0; //line goes from 0 to 3 (length 4); column goes from 1 to 3 (length 3)
59  while(1){
60  LPC_GPIO0->FIOSETL = OUTPUT_MASK; //output bits to 1
61  LPC_GPIO0->FIOCLRL = 1<<(SHIFT_TO_OUTPUT + line); //line output bit to 0
62  if(KeyCaught()){
63  int col = (LPC_GPIO0->FIOPIN & INPUT_MASK)>>SHIFT_TO_INPUT; //input bits(column bits) to col
64  if (col != NUMBER_OF_COLUMNS) //if key is pressed in last column then the col value is correct
65  col ^= VAL_TO_XOR; //if key is pressed in other columns this xor puts correct value in col
66  if (line == NUMBER_OF_LINES - 1){ //combination supported in last line(row) of keypad
67  if (CheckCombination()){
68  return COMBINATION_CODE;
69  }
70  }
